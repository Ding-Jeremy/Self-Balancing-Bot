\chapter*{Nomenclature}

\begin{longtable}{p{0.15\textwidth} p{0.65\textwidth} p{0.15\textwidth}}
\hline
\textbf{Symbol} & \textbf{Definition} & \textbf{Unit} \\
\hline
\endhead

$T_{max}$ & Couple maximal fourni par un moteur pas à pas à l'arrêt & \si{\newton\metre} \\
\hline
$T_p$ & Couple gravitationnel généré par la masse ponctuelle située à l'extrémité du levier & \si{\newton\metre} \\
\hline
$T_m$ & Couple moteur appliqué à une roue par un moteur pas à pas & \si{\newton\metre} \\
\hline
$T$ & Couple résultant appliqué au système par les deux moteurs d'entraînement & \si{\newton\metre} \\
\hline
$F_r$ & Force de réaction tangentielle appliquée au chariot par le contact roue-sol & \si{\newton} \\
\hline
$F_g$ & Force gravitationnelle appliquée au centre de masse du levier due au champ de pesanteur terrestre & \si{\newton} \\
\hline
$x$ & Position linéaire du chariot suivant l'axe de déplacement horizontal & \si{\metre} \\
\hline
$\theta$ & Angle d'inclinaison du levier par rapport à la verticale, défini comme l'angle entre l'axe longitudinal du levier passant par l'axe des roues et le centre de masse, et la direction du vecteur gravité & \si{\radian} \\
\hline
$l$ & Distance entre l'axe de rotation du levier (axe des roues) et son centre de masse & \si{\metre} \\
\hline
$r$ & Rayon extérieur d'une roue du chariot & \si{\metre} \\
\hline
$m_b$ & Masse du corps rigide du levier (hors roues et masse ponctuelle ajoutée) & \si{\kilogram} \\
\hline
$m_p$ & Masse ponctuelle fixée à l'extrémité du levier & \si{\kilogram} \\
\hline
$m_c$ & Masse du chariot & \si{\kilogram} \\
\hline
$m_w$ & Masse d'une roue du chariot & \si{\kilogram} \\
\hline
$\phi$ & Angle d'inclinaison de la position de l'IMU par rapport à la verticale, défini comme l'angle entre l'axe longitudinal du levier passant par l'axe des roues et l'IMU & \si{\radian} \\
\hline
$\rho$ & Distance entre l'IMU et l'axe des roues & \si{\metre} \\
\hline
$J_w$ & Moment d'inertie d'une roue autour de son axe de rotation & \si{\kilogram\metre\squared} \\
\hline
$J_b$ & Moment d'inertie du levier autour de son centre de masse & \si{\kilogram\metre\squared} \\
\hline
$g$ & Accélération gravitationnelle terrestre & \si{\metre\per\second\squared} \\
\hline
$V_{ref}$ & Tension de référence utilisée pour le réglage du courant du pilote moteur TMC2226 & \si{\volt} \\
\hline
$K_{pv},T_{dv},T_{iv}$ & Paramètre du régulateur de la boucle externe (vitesse) & - \\
\hline
$K_{pt},T_{dt},T_{it}$ & Paramètre du régulateur de la boucle interne (angle) & - \\
\hline
\end{longtable}
